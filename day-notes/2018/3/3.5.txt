func binary_search(array []int, key index)int{
	min:=0
	max:=len(array)-1
	for {
		mid:=(min+max)/2
		
		if key<array[mid]{
			max=mid-1
		}else if key>array[mid]{
			min=mid+1
		}else{
			return mid+1
		}
	}
	
	return -1
}
